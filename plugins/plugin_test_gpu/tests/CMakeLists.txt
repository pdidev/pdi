cmake_minimum_required(VERSION 3.10...3.25)
find_package(GTest 1.8.0 REQUIRED)

set(RUNTEST_DIR "${CMAKE_CURRENT_LIST_DIR}/../cmake/runtest-dir")

# Add the plugin path to PDI_PLUGIN_PATH
set_property(DIRECTORY "${CMAKE_CURRENT_SOURCE_DIR}" PROPERTY TEST_INCLUDE_FILE "${CMAKE_CURRENT_BINARY_DIR}/TestPath.cmake")
file(GENERATE OUTPUT "${CMAKE_CURRENT_BINARY_DIR}/TestPath.cmake"
     CONTENT "
set(PDI_PLUGIN_PATH \"\$ENV{PDI_PLUGIN_PATH}\")\n
if(\"x\${PDI_PLUGIN_PATH}x\" STREQUAL xx)\n
set(ENV{PDI_PLUGIN_PATH} \"\$<TARGET_FILE_DIR:gpu_tests>\")\n
else()\n
set(ENV{PDI_PLUGIN_PATH} \"\$<TARGET_FILE_DIR:gpu_tests>:\${PDI_PLUGIN_PATH}\")\n
endif()
"
)

add_executable(gpu_tests gpu_tests.cxx)
target_compile_features(gpu_tests PUBLIC cxx_std_17)
target_link_libraries(gpu_tests PDI::PDI_C GTest::GTest GTest::Main)
gtest_discover_tests(gpu_tests)
